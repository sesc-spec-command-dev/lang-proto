int fibonacci (int n) {
    if ((n == 1) || (n == 2)) {
        return 1;
    }

    return fibonacci(n-1) + fibonacci(n-2);
}

int cycle(int b) {
    while (fibonacci(b+3) > 5) {
        write(b);
    }

    return b;
}

int foo(int a) {
    int c;
    c = a*100;

    if (c > 1000) {
        write(c);
    } else {
        write("10");
    }
}

int main (int c) {
    write(fibonacci(c));

    return fibonacci(cycle(c));
}
